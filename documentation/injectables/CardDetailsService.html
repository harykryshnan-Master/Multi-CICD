  count: number = 0;
  incrementCount() {
    this.count++;
    this.notificationCount.next(this.count);
 
  }
 
  decreaseCount() {
    if (this.count < 1)
      return;
    this.count--;
 
    this.notificationCount.next(this.count);
 
 
  }
 
  resetCount() {
 
    this.count = 0;
